    \begin{algorithm}[H]
        \caption{Insertion sort}
        \begin{algorithmic}[1]
        \Procedure{Insertion-Sort}{$xs$}
        \For{$j\gets 2\ \mathbf{to}\ len(xs)$}
        \State $indice\gets xs[j]$
        \State $i\gets j-1$
        \While{$i > 0\ \mathbf{and}\ xs[i]>comodin$}
        \State $xs[i+1]\gets xs[i]$
        \State $i\gets i-1$
        \EndWhile
        \State $xs[i+1]\gets indice$
        \EndFor
        \EndProcedure
        \end{algorithmic}
    \end{algorithm}
